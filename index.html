# Directory structure for the project: # /presidential-actions-monitor # |-- app/ # | |-- __init__.py # | |-- routes.py # |-- scraper/ # | |-- __init__.py # | |-- scraper.py # |-- templates/ # | |-- index.html # |-- static/ # | |-- style.css # |-- app.py # |-- requirements.txt # |-- README.md # app.py: Main entry point for the Flask app from flask import Flask from app.routes import app_routes app = Flask(__name__) app.register_blueprint(app_routes) if __name__ == "__main__": app.run(debug=True) # /app/routes.py: Defines Flask routes from flask import Blueprint, render_template, jsonify from scraper.scraper import fetch_headlines app_routes = Blueprint("app_routes", __name__) BASE_URL = "https://www.whitehouse.gov/presidential-actions/" @app_routes.route("/") def home(): headlines = fetch_headlines(BASE_URL) return render_template("index.html", headlines=headlines) @app_routes.route("/api/headlines") def api_headlines(): headlines = fetch_headlines(BASE_URL) return jsonify(headlines) # /scraper/scraper.py: Handles web scraping import requests from bs4 import BeautifulSoup def fetch_headlines(base_url): headlines = [] page = 1 while True: url = f"{base_url}?page={page}" response = requests.get(url) if response.status_code != 200: break soup = BeautifulSoup(response.text, "html.parser") headline_tags = soup.find_all("h2", class_="action-title") if not headline_tags: break for tag in headline_tags: headlines.append(tag.text.strip()) page += 1 return headlines # /templates/index.html: HTML template Presidential Actions
